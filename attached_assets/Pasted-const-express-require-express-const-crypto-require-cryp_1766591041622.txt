const express = require("express");
const crypto = require("crypto");
const fs = require("fs");
const path = require("path");
const Database = require("better-sqlite3");

const app = express();

// Serve saved mp3 files
const UPLOAD_DIR = path.join(__dirname, "uploads");
if (!fs.existsSync(UPLOAD_DIR)) fs.mkdirSync(UPLOAD_DIR);
app.use("/uploads", express.static(UPLOAD_DIR));

// IMPORTANT: Trello webhook signature verification needs RAW body
app.use("/webhooks/trello", express.raw({ type: "*/*" }));

// --- ENV ---
const {
  TRELLO_KEY,
  TRELLO_TOKEN,
  TRELLO_API_SECRET,
  PUBLIC_WEBHOOK_URL,
  TRELLO_BOARD_SHORTLINK,
  TRELLO_LIST_INTAKE,
  TRELLO_LIST_UNRESOLVED,
  TRELLO_LIST_RESOLVED,
} = process.env;

// --- DB ---
const db = new Database("voicemails.db");
db.exec(`
  CREATE TABLE IF NOT EXISTS voicemails (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    trello_card_id TEXT UNIQUE,
    trello_list_id TEXT,
    title TEXT,
    description TEXT,
    status TEXT,
    caller TEXT,
    received_at TEXT,
    mp3_filename TEXT,
    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT DEFAULT (datetime('now'))
  );
`);

const upsertVoicemail = db.prepare(`
  INSERT INTO voicemails (trello_card_id, trello_list_id, title, description, status, caller, received_at, mp3_filename)
  VALUES (@trello_card_id, @trello_list_id, @title, @description, @status, @caller, @received_at, @mp3_filename)
  ON CONFLICT(trello_card_id) DO UPDATE SET
    trello_list_id=excluded.trello_list_id,
    title=excluded.title,
    description=excluded.description,
    status=excluded.status,
    caller=excluded.caller,
    received_at=excluded.received_at,
    mp3_filename=COALESCE(excluded.mp3_filename, voicemails.mp3_filename),
    updated_at=datetime('now');
`);

const setMp3 = db.prepare(`
  UPDATE voicemails
  SET mp3_filename = ?, updated_at=datetime('now')
  WHERE trello_card_id = ?;
`);

function statusFromListId(listId) {
  if (TRELLO_LIST_RESOLVED && listId === TRELLO_LIST_RESOLVED) return "RESOLVED";
  if (TRELLO_LIST_UNRESOLVED && listId === TRELLO_LIST_UNRESOLVED) return "UNRESOLVED";
  if (TRELLO_LIST_INTAKE && listId === TRELLO_LIST_INTAKE) return "NEW";
  // fallback if you haven’t set all list ids yet:
  if (TRELLO_LIST_INTAKE && listId === TRELLO_LIST_INTAKE) return "NEW";
  return "OTHER";
}

function parseCallerFromTitle(title) {
  // Example title: "Voicemail message from 'TRACI WAITE,Office - B (+140...)' [00:00:11]"
  const m = title.match(/from\s+"?([^"]+)"?/i);
  return m ? m[1].trim() : null;
}

function verifyTrelloSignature(req, rawBodyStr) {
  const sig = req.get("x-trello-webhook");
  if (!sig) return false;
  const content = rawBodyStr + PUBLIC_WEBHOOK_URL; // must match callbackURL exactly
  const digest = crypto.createHmac("sha1", TRELLO_API_SECRET).update(content).digest("base64");
  return digest === sig;
}

async function trelloGet(url) {
  const r = await fetch(url);
  if (!r.ok) {
    const text = await r.text().catch(() => "");
    throw new Error(`Trello API ${r.status}: ${text}`);
  }
  return r.json();
}

async function fetchCard(cardId) {
  const url =
    `https://api.trello.com/1/cards/${cardId}` +
    `?fields=name,desc,idList,dateLastActivity` +
    `&attachments=true&attachment_fields=id,name,mimeType,url` +
    `&key=${encodeURIComponent(TRELLO_KEY)}&token=${encodeURIComponent(TRELLO_TOKEN)}`;
  return trelloGet(url);
}

async function downloadAttachment(cardId, attachment) {
  // Uses Trello download endpoint with key/token (more reliable than attachment.url)
  const safeName = encodeURIComponent(attachment.name || "voicemail.mp3");
  const dl =
    `https://api.trello.com/1/cards/${cardId}/attachments/${attachment.id}/download/${safeName}` +
    `?key=${encodeURIComponent(TRELLO_KEY)}&token=${encodeURIComponent(TRELLO_TOKEN)}`;

  const r = await fetch(dl);
  if (!r.ok) throw new Error(`Download failed ${r.status}`);
  const buf = Buffer.from(await r.arrayBuffer());
  return buf;
}

function saveMp3ToDisk(buffer, originalName = "voicemail.mp3") {
  const safe = `${Date.now()}-${originalName.replace(/[^a-zA-Z0-9._-]/g, "_")}`;
  const outPath = path.join(UPLOAD_DIR, safe);
  fs.writeFileSync(outPath, buffer);
  return safe;
}

// --- Webhook verify endpoints (Trello pings these) ---
app.head("/webhooks/trello", (req, res) => res.sendStatus(200));
app.get("/webhooks/trello", (req, res) => res.sendStatus(200));

// --- Main Trello webhook receiver ---
app.post("/webhooks/trello", async (req, res) => {
  const rawBody = req.body.toString("utf8");

  if (!verifyTrelloSignature(req, rawBody)) {
    return res.status(401).send("Bad signature");
  }

  let payload;
  try {
    payload = JSON.parse(rawBody);
  } catch {
    return res.status(400).send("Bad JSON");
  }

  const action = payload?.action;
  const type = action?.type;

  // Identify cardId from common event shapes
  const cardId =
    action?.data?.card?.id ||
    action?.data?.card?.idShort || // sometimes present but not reliable
    null;

  // Some events store card in a different place:
  const cardIdAlt = action?.data?.card?.id;
  const finalCardId = cardIdAlt || cardId;

  if (!finalCardId) return res.sendStatus(200);

  try {
    const card = await fetchCard(finalCardId);

    // Only process if it’s one of the lists you care about.
    // Start with Intake only; once you set unresolved/resolved list ids, it’ll track those too.
    const watchedLists = new Set(
      [TRELLO_LIST_INTAKE, TRELLO_LIST_UNRESOLVED, TRELLO_LIST_RESOLVED].filter(Boolean)
    );

    if (watchedLists.size && !watchedLists.has(card.idList)) {
      return res.sendStatus(200);
    }

    const status = statusFromListId(card.idList);
    const caller = parseCallerFromTitle(card.name);

    upsertVoicemail.run({
      trello_card_id: card.id,
      trello_list_id: card.idList,
      title: card.name,
      description: card.desc || "",
      status,
      caller,
      received_at: card.dateLastActivity || null,
      mp3_filename: null,
    });

    // If an MP3 is attached, download & save it (idempotent: only saves when present)
    const mp3 = (card.attachments || []).find(a => {
      const name = (a.name || "").toLowerCase();
      const mime = (a.mimeType || "").toLowerCase();
      return name.endsWith(".mp3") || mime.includes("audio");
    });

    if (mp3) {
      // If we already saved an mp3 for this card, skip downloading again
      const existing = db
        .prepare("SELECT mp3_filename FROM voicemails WHERE trello_card_id = ?")
        .get(card.id);

      if (!existing?.mp3_filename) {
        const buf = await downloadAttachment(card.id, mp3);
        const savedName = saveMp3ToDisk(buf, mp3.name || "voicemail.mp3");
        setMp3.run(savedName, card.id);
        console.log(`[VOICEMAIL] Saved mp3 for card ${card.id}: ${savedName}`);
      }
    }

    console.log(`[WEBHOOK] ${type} -> processed card ${card.id} (${status})`);
    return res.sendStatus(200);
  } catch (err) {
    console.error("Webhook processing error:", err);
    // Return 200 so Trello doesn't repeatedly retry in a tight loop
    return res.sendStatus(200);
  }
});

// --- Helper: list Trello lists + IDs (so you can set env vars correctly) ---
app.get("/setup/trello-lists", async (req, res) => {
  try {
    const url =
      `https://api.trello.com/1/boards/${encodeURIComponent(TRELLO_BOARD_SHORTLINK)}/lists` +
      `?fields=name,id&key=${encodeURIComponent(TRELLO_KEY)}&token=${encodeURIComponent(TRELLO_TOKEN)}`;
    const lists = await trelloGet(url);
    res.json(lists);
  } catch (e) {
    res.status(500).json({ error: String(e) });
  }
});

// --- API for your frontend ---
app.get("/api/voicemails", (req, res) => {
  const status = req.query.status; // NEW / UNRESOLVED / RESOLVED / OTHER
  let rows;
  if (status) {
    rows = db.prepare("SELECT * FROM voicemails WHERE status = ? ORDER BY id DESC").all(status);
  } else {
    rows = db.prepare("SELECT * FROM voicemails ORDER BY id DESC").all();
  }
  res.json(rows);
});

// Simple page to view
app.get("/", (req, res) => {
  res.type("html").send(`
  <h2>Voicemail Mirror (Trello -> Replit)</h2>
  <p>Lists: <a href="/setup/trello-lists">/setup/trello-lists</a></p>
  <p>API: <a href="/api/voicemails">/api/voicemails</a></p>
  <p>Uploads: /uploads/&lt;filename&gt;</p>
  `);
});

const PORT = 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
